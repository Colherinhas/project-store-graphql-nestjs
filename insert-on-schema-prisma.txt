model Product {
  id           String           @id @unique @default(uuid())
  type         ProductsTypeEnum
  name         String
  // carts        CartProduct[]
  availability Boolean          @default(true)
}

model User {
  id        String         @id @unique @default(uuid())
  name      String
  email     String
  status    UserStatusEnum @default(ACTIVE)
  password  String
  // carts     Cart[]         @relation("UserCarts")
  createdAt DateTime       @default(now())
  updatedAt DateTime?      @updatedAt
  deletedAt DateTime?
}

enum ProductsTypeEnum {
  CLOTHING
  ELECTRONICS
  EXERCISING
  TOYS
}

enum UserStatusEnum {
  ACTIVE
  INACTIVE
  BANNED
}
